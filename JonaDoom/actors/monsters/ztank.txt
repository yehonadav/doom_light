ACTOR ZombieTank 21012
{
//$Category JonaDoom_Monsters
spawnid 217
painchance "Stunner", 32

    Health 600
    Painchance 50
    Speed 6
    Radius 27
    +NOBLOOD
    Height 72
    Mass 1000
    SeeSound "ZTank/see"
    AttackSound "ZTank/attack"
    PainSound "ZTank/pain"
    DeathSound "ZTank/death"
    DropItem Zombieman
    Obituary "%o was shot by a zombie in a tank."
    MONSTER 
    +FLOORCLIP
    Decal Bulletchip
    States
    {
    Spawn:
        ZTNK A 8 A_Look
        Loop
    See:
        ZTNK A 0 A_PlaySound ("ZTank/Run")
        ZTNK AABBCC 1 A_Chase
        Loop
    Missile:
        ZTNK A 1 A_FaceTarget
        ZTNK D 3 BRIGHT A_CustomBulletAttack (8, 5, 2, 2, "BulletPuff") 
        ZTNK A 1 A_FaceTarget
        ZTNK E 3 BRIGHT A_CustomBulletAttack (8, 5, 2, 2, "BulletPuff") 
        ZTNK A 0 A_Jump(128,1) 
        goto See 
        ZTNK A 0 A_CPosRefire
        goto Missile 
    Pain:
        ZTNK F 4 A_Pain
        ZTNK F 4
        Goto See
    Pain.Stunner:
       ZTNK F 2 A_ChangeFlag("NoPain", 1)
       ZTNK F 300 A_Pain
       ZTNK F 0 A_ChangeFlag("NoPain", 0)
       goto See       
    Death:
        TNT1 A 0 A_GiveToTarget("AwardItem", 6)
        ZTNK G 4
        ZTNK G 4 A_Scream
        ZTNK H 8 A_SpawnItemEx("ZombieTankGun", -40, 0, 30, 12, 0, 0, 64, 128)
        ZTNK I 8 A_SpawnItemEx("ZombieTankGun", 40, 0, 30, 12, 0, 0, 192, 128)
        ZTNK J 8 A_CustomMissile ("ZombieTankExplosion", 48, 0, 0)
        ZTNK K 8 A_Fall
        ZTNK L 8
        ZTNK M 8 A_SpawnItemEx("ZTankSmoke", 0, 0, 48, 0, 0, 0, 0, 128)
        ZTNK N 8
        ZTNK O 3000
        ZTNK OOOOOOOOOOOO 2 A_FadeOut(0.1)        
        Stop
    }
}
