Actor "LavaParticle"
{
    +BLOODLESSIMPACT
    +BRIGHT
    +CANNOTPUSH
    +DONTSPLASH
    +DROPOFF
    +FORCEXYBILLBOARD
	+HEXENBOUNCE
    +MISSILE
    +NOBLOCKMAP
    +NOTELEPORT
    +PAINLESS
    +RIPPER
        
    Radius              1
    Height              2
    Speed               0
    Gravity             0.25
    Scale               0.04
    BounceFactor        0.5
    WallBounceFactor    0.5
        
    States
    {
    "Spawn":
        PART A 0
        PART A 0 A_JumpIf(waterlevel > 0, 2)
        PART A 1 A_FadeOut(0.05)
        loop
        
        TNT1 A 0
        stop
    }
}
